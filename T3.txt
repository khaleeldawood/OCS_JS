function tellFortune(jobTitle, location, partnerName, numberOfKids) {
  return `You will be a ${jobTitle} in ${location}, and married to ${partnerName} with ${numberOfKids} kids.`;
}

console.log(tellFortune('software engineer', 'Jordan', 'Alice', 3));


function calculateDogAge(puppyAge) {
  return `Your doggie is ${puppyAge * 7} years old in dog years!`;
}

console.log(calculateDogAge(1));
