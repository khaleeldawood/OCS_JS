    <script>
        // Task 1
        function startHover() {
            const box = document.getElementById('hoverBox');
            box.style.backgroundColor = 'yellow';
            box.style.fontWeight = 'bold';
        }
        function endHover() {
            const box = document.getElementById('hoverBox');
            box.style.backgroundColor = 'transparent';
            box.style.fontWeight = 'normal';
        }

        // Task 2
        function showFlag() {
            const country = document.getElementById('countrySelect').value;
            const display = document.getElementById('flagDisplay');
            // Using Emojis as image files are not provided
            if (country === 'Jordan') display.innerText = "ðŸ‡¯ðŸ‡´";
            else if (country === 'USA') display.innerText = "ðŸ‡ºðŸ‡¸";
            else if (country === 'Norway') display.innerText = "ðŸ‡³ðŸ‡´";
            else display.innerText = "";
        }

        // Task 3
        function checkPass() {
            const p1 = document.getElementById('p1').value;
            const p2 = document.getElementById('p2').value;
            
            // Check length (min 6 based on screenshot context)
            if (p1.length > 0 && p1.length < 6) {
                document.getElementById('p1Error').style.display = 'inline';
            } else {
                document.getElementById('p1Error').style.display = 'none';
            }

            // Check match
            if (p2.length > 0 && p1 !== p2) {
                document.getElementById('p2Error').style.display = 'inline';
            } else {
                document.getElementById('p2Error').style.display = 'none';
            }
        }

        // Task 4
        function toggleText(e) {
            e.preventDefault();
            const textDiv = document.getElementById('loremText');
            const link = document.getElementById('toggleLink');
            
            if (textDiv.style.display === 'none' || textDiv.style.display === '') {
                textDiv.style.display = 'block';
                link.innerText = 'Hide';
            } else {
                textDiv.style.display = 'none';
                link.innerText = 'Show';
            }
        }

        // Task 5
        function updateEditor() {
            const box = document.getElementById('editorBox');
            
            // Font Family
            box.style.fontFamily = document.getElementById('fontFamily').value;
            // Font Size
            box.style.fontSize = document.getElementById('fontSize').value;
            // Italic
            box.style.fontStyle = document.getElementById('isItalic').checked ? 'italic' : 'normal';
            // Bold
            box.style.fontWeight = document.getElementById('isBold').checked ? 'bold' : 'normal';
            // Underline
            box.style.textDecoration = document.getElementById('isUnderline').checked ? 'underline' : 'none';
        }

        // Task 6
        // Changes color every 1 second
        setInterval(function() {
            const box = document.getElementById('animBox');
            const colors = ['blue', 'red', 'green', 'orange', 'purple', 'teal'];
            const randomColor = colors[Math.floor(Math.random() * colors.length)];
            box.style.backgroundColor = randomColor;
        }, 1000);

    </script>
