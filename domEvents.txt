<script>
    function startHover() {
        const box = document.getElementById('hoverBox');
        box.style.backgroundColor = 'yellow';
        box.style.fontWeight = 'bold';
    }
    function endHover() {
        const box = document.getElementById('hoverBox');
        box.style.backgroundColor = 'white';
        box.style.fontWeight = 'normal';
    }

    function showFlag() {
        const country = document.getElementById('countrySelect').value;
        const display = document.getElementById('flagDisplay');
        if(country === 'norway') display.innerText = "ðŸ‡³ðŸ‡´";
        else if(country === 'jordan') display.innerText = "ðŸ‡¯ðŸ‡´";
        else if(country === 'usa') display.innerText = "ðŸ‡ºðŸ‡¸";
        else display.innerText = "";
    }

    function checkPass() {
        const p1 = document.getElementById('p1');
        const p2 = document.getElementById('p2');
        const p1Err = document.getElementById('p1Error');
        const p2Err = document.getElementById('p2Error');
        if(p1.value.length < 6 && p1.value.length > 0) {
            p1Err.style.display = 'inline';
        } else {
            p1Err.style.display = 'none';
        }

        if(p2.value !== p1.value && p2.value.length > 0) {
            p2Err.style.display = 'inline';
        } else {
            p2Err.style.display = 'none';
        }
    }

    function toggleText(e) {
        e.preventDefault();
        const text = document.getElementById('loremText');
        const link = e.target;
        if (text.style.display === 'none') {
            text.style.display = 'block';
            link.innerText = 'Hide';
        } else {
            text.style.display = 'none';
            link.innerText = 'Show';
        }
    }

    function formatDoc(cmd, value = null) {
        if(value) {
            document.execCommand(cmd, false, value);
        } else {
            document.execCommand(cmd);
        }
    }

    setInterval(function() {
        const box = document.getElementById('animBox');
        const colors = ['blue', 'green', 'red', 'orange', 'purple'];
        const randomColor = colors[Math.floor(Math.random() * colors.length)];
        box.style.backgroundColor = randomColor;
    }, 1000);
</script>
